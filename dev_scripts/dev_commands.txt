RUN ALL TESTS
----------------------------------------------------------------------------------------------------------------------------------------------------
    python3 -m pytest -vv --show-capture=stdout --cov-report=html --cov-report=xml --junitxml=junit_report.xml --cov=./ .
----------------------------------------------------------------------------------------------------------------------------------------------------


RUN UNIT TESTS
----------------------------------------------------------------------------------------------------------------------------------------------------
    python3 -m pytest -vv --show-capture=stdout --cov-report=html --cov-report=xml --junitxml=junit_report.xml --cov=./ tests/
----------------------------------------------------------------------------------------------------------------------------------------------------


RUN E2E TESTS
----------------------------------------------------------------------------------------------------------------------------------------------------
    python3 -m pytest -vv --show-capture=stdout --cov-report=html --cov-report=xml --junitxml=junit_report.xml --cov=./ e2e-tests/
----------------------------------------------------------------------------------------------------------------------------------------------------


RUN FAILED TESTS
----------------------------------------------------------------------------------------------------------------------------------------------------
python3 -m pytest -vv --show-capture=stdout --cov-report=html --cov-report=xml --junitxml=junit_report.xml --cov=./ tests/ --last-failed
----------------------------------------------------------------------------------------------------------------------------------------------------


RUN SPECIFIC TESTS
----------------------------------------------------------------------------------------------------------------------------------------------------
    python3 -m pytest -vv --show-capture=stdout --cov-report=html --cov-report=xml --junitxml=junit_report.xml --cov=./ -k <TEST_FILE_NAME_OR_TEST_NAME>
----------------------------------------------------------------------------------------------------------------------------------------------------
    eg:
        python3 -m pytest -vv --show-capture=stdout --cov-report=html --cov-report=xml --junitxml=junit_report.xml --cov=./ -k utils_translation_test
        python3 -m pytest -vv --show-capture=stdout --cov-report=html --cov-report=xml --junitxml=junit_report.xml --cov=./ -k test_choose_right_tracker_keys_disc_enabled
----------------------------------------------------------------------------------------------------------------------------------------------------
